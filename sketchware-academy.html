Google Apps Script Deployment Guide

1
📋 Copy the Script

Copy the complete Apps Script code from one of the API tabs below.

2
🌐 Go to script.google.com

Open script.google.com and sign in with your Google account.

3
📁 Create New Project

Click + New project. Delete any default code and paste your script.

4
💾 Save the Project

Click the save icon (💾) or press Ctrl + S. Name your project (e.g., "Sketchware Upload API").

5
🚀 Deploy as Web App

Click Deploy → New deployment → Click the settings icon (⚙️) → Select Web app.

⚙️ Deployment Settings:
• Execute as: Me (your email)
• Who has access: Anyone (for testing from Sketchware)
• Click Deploy
6
🔐 Authorize the App

Google will ask for permissions. Click Review permissions → Choose your account → Allow.

7
📎 Copy Web App URL

After deployment, copy the URL (looks like: https://script.google.com/macros/s/.../exec).

https://script.google.com/macros/s/AKfycbxxxxxx/exec
8
📱 Use in Sketchware

Paste this URL in your Sketchware HTTP Request component.

Re-deploying After Changes

If you modify your script, you need to create a new version:

1️⃣ Edit your code in Apps Script editor
2️⃣ Click Deploy → Manage deployments
3️⃣ Click ✏️ Edit (pencil icon) next to your deployment
4️⃣ Select New version → Update → Copy new URL

File Upload API (Drive)

This script uploads files to your Google Drive with file type validation.

Complete Script

// ============================================
// 📁 Google Apps Script - Sketchware File Upload API
// ============================================

function doPost(e) {
  const res = {};
  try {
    const fileBlob = e.parameter.file;
    if (!fileBlob) {
      throw new Error('No file uploaded. Use parameter name: "file"');
    }
    
    // 🔧 REPLACE WITH YOUR FOLDER ID
    const folder = DriveApp.getFolderById("YOUR_FOLDER_ID_HERE");
    
    const file = folder.createFile(fileBlob);
    const fileId = file.getId();
    const fileName = file.getName();
    const mimeType = file.getMimeType();
    const fileSize = file.getSize();
    
    // File type validation (change as needed)
    if (!mimeType.startsWith('image/')) {
      throw new Error('❌ Only image files are allowed');
    }
    
    const fileUrl = `https://drive.google.com/uc?export=download&id=${fileId}`;
    
    res.success = true;
    res.message = "File uploaded successfully";
    res.fileId = fileId;
    res.fileName = fileName;
    res.fileUrl = fileUrl;
    res.mimeType = mimeType;
    res.fileSize = fileSize;
    res.folderId = "YOUR_FOLDER_ID_HERE";
    
  } catch (error) {
    res.success = false;
    res.error = error.toString();
  }
  
  return ContentService
    .createTextOutput(JSON.stringify(res))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet() {
  return ContentService.createTextOutput(JSON.stringify({
    message: "✅ Sketchware upload endpoint active. Use POST with multipart/form-data, field name: 'file'"
  })).setMimeType(ContentService.MimeType.JSON);
}
Customization:
• Replace YOUR_FOLDER_ID_HERE with your actual Drive folder ID
• Change file validation: if (!mimeType.startsWith('image/')) → if (mimeType !== 'application/pdf') for PDF only
• Remove validation for "any file"

CRUD API (Google Sheets)

Create, Read, Update, Delete data from Google Sheets via Sketchware.

Complete CRUD Script

/**
 * CRUD API for Sketchware - Google Sheets Database
 */

// 🔧 CONFIGURATION - REPLACE WITH YOUR VALUES
const SS_ID = "YOUR_SPREADSHEET_ID_HERE";
const SHEET_NAME = "Sheet1";

function doPost(e) {
  const p = e.parameter;
  const ss = SpreadsheetApp.openById(SS_ID).getSheetByName(SHEET_NAME);
  try {
    const action = p.action;
    
    if (action === 'create') return createRecord(ss, p);
    if (action === 'read') return readRecords(ss, p);
    if (action === 'update') return updateRecord(ss, p);
    if (action === 'delete') return deleteRecord(ss, p);
    
    return sendResponse({ error: "Invalid action. Use: create, read, update, delete" });
  } catch(e) { 
    return sendResponse({ error: e.toString() }); 
  }
}

function createRecord(ss, p) {
  const name = p.name || "";
  const email = p.email || "";
  const phone = p.phone || "";
  const status = p.status || "active";
  ss.appendRow([new Date(), name, email, phone, status]);
  return sendResponse({ success: true, message: "Record created" });
}

function readRecords(ss, p) {
  const data = ss.getDataRange().getValues();
  if (data.length <= 1) {
    return sendResponse({ success: true, data: [], message: "No records found" });
  }
  const headers = data[0];
  const records = [];
  for (let i = 1; i < data.length; i++) {
    const record = {};
    headers.forEach((header, idx) => { record[header] = data[i][idx]; });
    records.push(record);
  }
  const search = p.search;
  if (search) {
    const filtered = records.filter(r => 
      JSON.stringify(r).toLowerCase().includes(search.toLowerCase())
    );
    return sendResponse({ success: true, data: filtered, total: filtered.length });
  }
  return sendResponse({ success: true, data: records, total: records.length });
}

function updateRecord(ss, p) {
  const rowNum = parseInt(p.row);
  if (isNaN(rowNum) || rowNum < 2) {
    return sendResponse({ error: "Valid row number required (row >= 2)" });
  }
  if (p.name) ss.getRange(rowNum, 2).setValue(p.name);
  if (p.email) ss.getRange(rowNum, 3).setValue(p.email);
  if (p.phone) ss.getRange(rowNum, 4).setValue(p.phone);
  if (p.status) ss.getRange(rowNum, 5).setValue(p.status);
  return sendResponse({ success: true, message: `Row ${rowNum} updated` });
}

function deleteRecord(ss, p) {
  const rowNum = parseInt(p.row);
  if (isNaN(rowNum) || rowNum < 2) {
    return sendResponse({ error: "Valid row number required (row >= 2)" });
  }
  ss.deleteRow(rowNum);
  return sendResponse({ success: true, message: `Row ${rowNum} deleted` });
}

function sendResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet() {
  return sendResponse({ 
    message: "✅ CRUD API Active", 
    actions: ["create", "read", "update", "delete"]
  });
}

Sketchware Setup - HTTP Request Blocks

File Upload Setup

📦 When Button Click → [HTTP Request]
  ├─ Method: POST
  ├─ URL: [Your Apps Script URL]
  ├─ Add Parameter: file → Type: File/Multipart → Value: [FilePicker Selected File]
  └─ On Response → Show Toast / Parse JSON

CRUD API Setup

📦 Create Record
  ├─ Method: POST
  ├─ Parameters:
    ├─ action = "create"
    ├─ name = [EditText Name]
    ├─ email = [EditText Email]
    └─ phone = [EditText Phone]
📖 Read Records
  ├─ Method: POST
  ├─ Parameters:
    ├─ action = "read"
    └─ search = [EditText Search] (optional)
✏️ Update Record
  ├─ Method: POST
  ├─ Parameters:
    ├─ action = "update"
    ├─ row = [Row Number]
    └─ name = [New Name] (optional)
🗑️ Delete Record
  ├─ Method: POST
  ├─ Parameters:
    ├─ action = "delete"
    └─ row = [Row Number]
Pro Tip:
Always use POST method with application/x-www-form-urlencoded or multipart/form-data for file uploads. Parse the JSON response using Sketchware's "JSON Object" component.

Response Handling in Sketchware

📦 HTTP Request → On Response
  ├─ JSON Object → Set text = response
  ├─ Get value: success → If true:
    ├─ Show Toast: "Upload successful!"
    ├─ Get value: fileUrl → Save to variable
  └─ Else: Show Toast: "Error: " + error
